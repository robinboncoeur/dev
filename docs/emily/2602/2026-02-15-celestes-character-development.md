Not dramatic, not storm-at-midnight, just… Tuesday. The kind of day that already smelled like hot bitumen and cheap coffee by nine in the morning.

“How was England?” I asked, when the silence started to feel like a third person in the car.

Useful. That was Mum for *I tore the last bits of something out by the roots.*

I watched her profile at the next set of lights. The little lines around her mouth looked different, somehow. Less like strain, more like decisions that had finally finished congealing.

The light went green. She didn’t move.

Cars stacked up behind us. No one honked yet. Australia could be polite like that, right up to the second it wasn’t.

Mum looked at the intersection, then at me.

“I’m Australian now,” she said. “On paper and in my bones. I told him that. I laid it out: we could apply for a partner visa, we could build something *here*, but he’d have to actually choose. She let the car roll forward then, easing us through the lights as if we had all the time in the world.

“Final order came through last month,” she added. “Decree absolute. I’m done waiting for a man to decide who he is.”

The words went straight down my spine and sat there.

Not because they were shocking. Because they weren’t. They were the clean, sharp version of what I’d watched my whole childhood: Mum making decisions in the gap where another adult should have been standing.

If you don’t move, someone you love bleeds for you.

I’d felt that sentence in my body for years without having the language. Hearing her say she was done with it… it was like someone finally turned the lights on in a room I’d been walking through in the dark.

“I’m glad,” I said. It came out rougher than I meant. “That you chose.”

She glanced at me, then back at the road.

“So am I,” she said. “Took me long enough.”

We drove the rest of the way with the air rushing in and the radio doing that soft, background murmur. I watched the suburbs slide past and thought, very clearly:

*Nobody is going to bleed for my decisions. Not Mum. Not me. Not anyone I love.*

Which was a noble thought.

And, as I would realise later, also a dangerous one — because once you’ve decided you won’t let people bleed for you, it’s a very short step to deciding you’ll steer their lives for them instead.

```python
from datetime import datetime
```

```python
OUTPUT_DIR = Path("/path/to/your/mkdocs/docs/chat")

        return False
    return True

        if not title_matches(title):
            continue

        created = convo.get("create_time")
        if created:
            dt = datetime.fromtimestamp(created)
            date_str = dt.strftime("%Y-%m-%d")
        else:
            date_str = "1970-01-01"

        slug = slugify(title)
        filename = f"{date_str}-{slug}.md"
        out_path = OUTPUT_DIR / filename

        # Avoid accidentally appending to old versions
        lines = []

        lines.append(f'title: "{title}"')
        lines.append(f"date: {date_str}")
        lines.append("---\n")
        lines.append(f"# {title}\n")

        mapping = convo.get("mapping", {})
        # messages are usually in mapping.values(); but order may be given by "children"/"id".
        # Simplest approach: sort by create_time if available.
        messages = []
        for node in mapping.values():
            msg = node.get("message")
            if not msg:
                continue
            ts = msg.get("create_time", 0)
            messages.append((ts, msg))

        messages.sort(key=lambda x: x[0])

            if not text:

            if author == "user":
                lines.append(f"**{ROBYN_NAME}:**")
            elif author == "assistant":
                lines.append(f"**{ASSISTANT_NAME}:**")
            else:
                label = author or "system"
                lines.append(f"**{label}:**")

            lines.append("")
            lines.append(text)
            lines.append("")

        out_path.write_text("\n".join(lines), encoding="utf-8")
        print(f"Wrote {out_path}")

if __name__ == "__main__":
    main()
```

- handles both `str` and `dict` items,
- looks for likely keys like `"text"` or `"content"`,
- falls back to `str(...)` if needed.

### Updated `aJConv` (just paste over your current version)

```python
import json
import re
from pathlib import Path
from datetime import datetime

# ==== CONFIG ====

BASE_DIR = Path(__file__).resolve().parent

# conversations.json should be in the same folder as this script
EXPORT_PATH = BASE_DIR / "conversations.json"

# where to put the markdown files (change this to your MkDocs docs/chat if you like)
OUTPUT_DIR = BASE_DIR / "out_md"

ROBYN_NAME = "Robyn"
ASSISTANT_NAME = "Emily"

TITLE_CONTAINS = None   # e.g. "Celeste" to only export those; or None for all
TITLE_REGEX = None      # e.g. r"^2602\d{2}" ; or None for all

# ==== HELPERS ====


def slugify(title: str) -> str:
    slug = re.sub(r"[^a-zA-Z0-9]+", "-", title).strip("-").lower()
    return slug or "untitled"


def title_matches(title: str) -> bool:
    if TITLE_CONTAINS and TITLE_CONTAINS.lower() not in title.lower():
        return False
    if TITLE_REGEX and not re.search(TITLE_REGEX, title):
        return False
    return True


def extract_text_from_parts(parts):
    """Handle both string and dict parts from the export."""
    texts = []
    for p in parts:
        # Plain string
        if isinstance(p, str):
            texts.append(p)
            continue

        # Dict-style part
        if isinstance(p, dict):
            # Most common: {"text": "...", "content_type": "text"}
            if "text" in p and isinstance(p["text"], str):
                texts.append(p["text"])
                continue

            # Sometimes "content" might hold the text
            if "content" in p:
                c = p["content"]
                if isinstance(c, str):
                    texts.append(c)
                    continue
                # if it's a list of nested segments
                if isinstance(c, list):
                    inner = []
                    for seg in c:
                        if isinstance(seg, str):
                            inner.append(seg)
                        elif isinstance(seg, dict) and "text" in seg:
                            inner.append(str(seg["text"]))
                    if inner:
                        texts.append("\n".join(inner))
                        continue

        # Fallback: just stringify
        texts.append(str(p))

    return "\n\n".join(texts).strip()


def load_conversations():
    raw = EXPORT_PATH.read_text(encoding="utf-8")
    data = json.loads(raw)

    # Case 1: export is a dict with a "conversations" key
    if isinstance(data, dict):
        convs = data.get("conversations")
        if isinstance(convs, list):
            return convs
        # fallback: maybe the dict itself is one convo
        return [data]

    # Case 2: export is already a list of conversations
    if isinstance(data, list):
        return data

    raise ValueError("Unexpected JSON top-level structure; expected list or dict.")


def main():
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

    conversations = load_conversations()

    for convo in conversations:
        title = convo.get("title") or "ChatGPT Conversation"
        if not title_matches(title):
            continue

        created = convo.get("create_time")
        if created:
            try:
                dt = datetime.fromtimestamp(created)
                date_str = dt.strftime("%Y-%m-%d")
            except (TypeError, OSError):
                date_str = "1970-01-01"
        else:
            date_str = "1970-01-01"

        slug = slugify(title)
        filename = f"{date_str}-{slug}.md"
        out_path = OUTPUT_DIR / filename

        lines = []
        lines.append("---")
        lines.append(f'title: "{title}"')
        lines.append(f"date: {date_str}")
        lines.append("---\n")
        lines.append(f"# {title}\n")

        mapping = convo.get("mapping", {})

        messages = []
        for node in mapping.values():
            msg = node.get("message")
            if not msg:
                continue
            ts = msg.get("create_time") or 0
            messages.append((ts, msg))

        # sort chronologically
        messages.sort(key=lambda x: x[0])

        for _, message in messages:
            author = message.get("author", {}).get("role")
            parts = message.get("content", {}).get("parts", [])
            if not parts:
                continue

            text = extract_text_from_parts(parts)
            if not text:
                continue

            if author == "user":
                lines.append(f"**{ROBYN_NAME}:**")
            elif author == "assistant":
                lines.append(f"**{ASSISTANT_NAME}:**")
            else:
                label = author or "system"
                lines.append(f"**{label}:**")

            lines.append("")
            lines.append(text)
            lines.append("")

        out_path.write_text("\n".join(lines), encoding="utf-8")
        print(f"Wrote {out_path}")


if __name__ == "__main__":
    main()
```
